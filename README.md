# DatabaseProject
## SQL DDL
CREATE TABLE TEAM (
 TeamName VARCHAR(50) PRIMARY KEY,
 League VARCHAR(50) NOT NULL
);

CREATE TABLE COACH (
 CoachID INT PRIMARY KEY,
 CoachName VARCHAR(50) NOT NULL,
 TeamName VARCHAR(50),
 FOREIGN KEY (TeamName) REFERENCES TEAM(TeamName)
);

CREATE TABLE PLAYER (
 PlayerID INT PRIMARY KEY,
 PlayerNumber INT NOT NULL,
 GamesPlayed INT DEFAULT 0,
 TeamName VARCHAR(50),
 FOREIGN KEY (TeamName) REFERENCES TEAM(TeamName),
 UNIQUE (TeamName, PlayerNumber)
);

CREATE TABLE PLAY_AGAINST (
 TeamA VARCHAR(50) NOT NULL,
 TeamB VARCHAR(50) NOT NULL,
 PRIMARY KEY (TeamA, TeamB),
 FOREIGN KEY (TeamA) REFERENCES TEAM(TeamName),
 FOREIGN KEY (TeamB) REFERENCES TEAM(TeamName)
);
CREATE TABLE PAYMENT (
 CheckNumber INT PRIMARY KEY,
 Department VARCHAR(50),
 TeamName VARCHAR(50),
 FOREIGN KEY (TeamName) REFERENCES TEAM(TeamName)
);

CREATE TABLE MANAGER (
 ManagerID INT PRIMARY KEY,
 Role VARCHAR(50)
);

CREATE TABLE FUNDS (
 FundID INT PRIMARY KEY,
 ManagerID INT NOT NULL,
 Amount DECIMAL(10,2) NOT NULL,
 FOREIGN KEY (ManagerID) REFERENCES MANAGER(ManagerID)
);

CREATE TABLE OWNER (
 OwnerName VARCHAR(50) PRIMARY KEY
);

CREATE TABLE SPONSOR (
 SponsorName VARCHAR(50) PRIMARY KEY,
 SponsorType VARCHAR(50)
);

CREATE TABLE GIVEMONEY (
 OwnerName VARCHAR(50) NOT NULL,
 SponsorName VARCHAR(50) NOT NULL,
 FundID INT NOT NULL,
 Amount DECIMAL(10,2) NOT NULL,
 PRIMARY KEY (OwnerName, SponsorName, FundID),
 FOREIGN KEY (OwnerName) REFERENCES OWNER(OwnerName),
 FOREIGN KEY (SponsorName) REFERENCES SPONSOR(SponsorName),
 FOREIGN KEY (FundID) REFERENCES FUNDS(FundID)
);
